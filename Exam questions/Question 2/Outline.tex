ferent pictures of quantum mechanics. (Heisenberg and Schr¨odinger)}

\underline{Unitary time evolution}

\begin{itemize}
    \item $\ket{\alpha,t_0; t} = \hat{\mathcal{U}}(t,t_0) \ket{\alpha, t_0}$
    \item Properties of $\hat{\mathcal{U}}(t,t_0)$:
    \begin{itemize}
        \item Prob. must be conserved $\Rightarrow$ $\hat{\mathcal{U}}$ is unitary
        \item Composition property: $\hat{\mathcal{U}}(t_2,t_0) = \hat{\mathcal{U}}(t_2,t_1)\hat{\mathcal{U}}(t_1,t_0)$
        \item $\hat{\mathcal{U}}(t_0,t_0) = \hat{I}$
    \end{itemize}
    \item Infinitesimal time evolution: $\hat{\mathcal{U}}(t_0 + dt, t_0) = \hat{1} - i\hat{\Omega} dt$ ($\Omega $ is Hermitian)
    \item Hamiltonian generates time evolution: $\hat{\mathcal{U}}(t_0 + dt, t_0) = \hat{1} - \frac{i}{\hbar} \hat{H}(t_0) dt$
    \item Schr¨odinger equation: $i\hbar \frac{\partial}{\partial t} \ket{\alpha, t_0; t} = \hat{H}(t) \ket{\alpha, t_0; t}$
\end{itemize}

\underline{Heisenberg vs. Schrödinger picture}

\begin{itemize}
    \item Possible because time evolution is unitary
    \item Schrödinger picture:
    \begin{itemize}
        \item States are time-dependent: $\ket{\alpha, t_0; t} = \hat{\mathcal{U}}(t,t_0) \ket{\alpha, t_0}$
        \item Operators are time-independent: $\hat{A}_S = \hat{A}$
        \item Expectation value: $\bra{\alpha, t_0; t} \hat{A}_S \ket{\alpha, t_0; t}$
    \end{itemize}
    \item Heisenberg picture:
    \begin{itemize}
        \item States are time-independent: $\ket{\alpha}_H = \ket{\alpha, t_0}$
        \item Operators are time-dependent: $\hat{A}_H(t) = \hat{\mathcal{U}}^\dagger(t,t_0) \hat{A}_S \hat{\mathcal{U}}(t,t_0)$
        \item Expectation value: $\bra{\alpha}_H \hat{A}_H(t) \ket{\alpha}_H$
    \end{itemize}
    \item unitary transformation: $\hat{A}_H(t) = \hat{\mathcal{U}}^\dagger(t,t_0) \hat{A}_S \hat{\mathcal{U}}(t,t_0)$
    \item \textcolor{red}{Equation of motion for operators in Heisenberg picture}
\end{itemize}
